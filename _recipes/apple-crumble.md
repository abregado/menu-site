  - item: apple
    amount: 750
    unit: g
    prep: peeled and sliced
  - item: flour
    amount: 120
    unit: g
    prep: sifted
  - item: butter
    amount: 113
    unit: g
    prep: cold, cubed
  - item: brown-sugar
    amount: 200
    unit: g
  - item: cinnamon
    amount: 5
    unit: g
  - item: nutmeg
    amount: 2
    unit: g
  - item: salt
    amount: 1
    unit: g
instructions:
  - Preheat oven to 190°C (375°F)
  - Prepare the ingredients
  - Toss apples with cinnamon
  - Place apples in baking dish
  - Mix flour, sugar, and butter until crumbly
  - Sprinkle over apples
  - Bake for 35-40 minutes until golden
